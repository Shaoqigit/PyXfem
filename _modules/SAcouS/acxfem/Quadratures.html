def get_quadrature_points_weights(n: int, dim: int):
  """Get quadrature points and weights
  args:
    n: int, number of quadrature points
    dim: int, dimension of the quadrature points
    """
  match dim:
    case 1:
      return gauss_legendre_1d_points(n), gauss_legendre_1d_weights(n)
    case 2:
      return gauss_legendre_2d_tri_points(n), gauss_legendre_2d_tri_weights(n)
    case 3:
      return gauss_legendre_3d_tetra_points(
          n), gauss_legendre_3d_tetra_weights(n)
    case 4:
      return gauss_legendre_3d_hex_points(n), None
    case _:
      raise NotImplementedError("Not implemented yet")
